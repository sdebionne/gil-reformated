251  std::size_t read(byte_t *data, std::size_t count) {
252  std::streamsize cr = 0;
253 
254  do {
255  _in.peek();
256  std::streamsize c = _in.readsome(reinterpret_cast<char *>(data),
257  static_cast<std::streamsize>(count));
258 
259  count -= static_cast<std::size_t>(c);
260  data += c;
261  cr += c;
262 
263  } while (count && _in);
264 
265  return static_cast<std::size_t>(cr);
266  }
