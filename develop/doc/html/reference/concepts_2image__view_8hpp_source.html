111 template <typename View> struct RandomAccessNDImageViewConcept {
112  void constraints() {
113  gil_function_requires<Regular<View>>();
114 
115  using value_type = typename View::value_type;
116  using reference = typename View::reference; // result of dereferencing
117  using pointer = typename View::pointer;
118  using difference_type =
119  typename View::difference_type; // result of
120  // operator-(1d_iterator,1d_iterator)
121  using const_t =
122  typename View::const_t; // same as this type, but over const values
123  using point_t = typename View::point_t; // N-dimensional point
124  using locator = typename View::locator; // N-dimensional locator
125  using iterator = typename View::iterator;
126  using const_iterator = typename View::const_iterator;
127  using reverse_iterator = typename View::reverse_iterator;
128  using size_type = typename View::size_type;
129  static const std::size_t N = View::num_dimensions;
130 
131  gil_function_requires<RandomAccessNDLocatorConcept<locator>>();
132  gil_function_requires<
133  boost_concepts::RandomAccessTraversalConcept<iterator>>();
134  gil_function_requires<
135  boost_concepts::RandomAccessTraversalConcept<reverse_iterator>>();
136 
137  using first_it_type = typename View::template axis<0>::iterator;
138  using last_it_type = typename View::template axis<N - 1>::iterator;
139  gil_function_requires<
140  boost_concepts::RandomAccessTraversalConcept<first_it_type>>();
141  gil_function_requires<
142  boost_concepts::RandomAccessTraversalConcept<last_it_type>>();
143 
144  // static_assert(typename
145  // std::iterator_traits<first_it_type>::difference_type, typename
146  // point_t::template axis<0>::coord_t>::value, "");
147  // static_assert(typename
148  // std::iterator_traits<last_it_type>::difference_type, typename
149  // point_t::template axis<N-1>::coord_t>::value, "");
150 
151  // point_t must be an N-dimensional point, each dimension of which must have
152  // the same type as difference_type of the corresponding iterator
153  gil_function_requires<PointNDConcept<point_t>>();
154  static_assert(point_t::num_dimensions == N, "");
155  static_assert(
156  std::is_same<
157  typename std::iterator_traits<first_it_type>::difference_type,
158  typename point_t::template axis<0>::coord_t>::value,
159  "");
160  static_assert(
161  std::is_same<
162  typename std::iterator_traits<last_it_type>::difference_type,
163  typename point_t::template axis<N - 1>::coord_t>::value,
164  "");
165 
166  point_t p;
167  locator lc;
168  iterator it;
169  reverse_iterator rit;
170  difference_type d;
171  detail::initialize_it(d);
172  ignore_unused_variable_warning(d);
173 
174  View(p, lc); // view must be constructible from a locator and a point
175 
176  p = view.dimensions();
177  lc = view.pixels();
178  size_type sz = view.size();
179  ignore_unused_variable_warning(sz);
180  bool is_contiguous = view.is_1d_traversable();
181  ignore_unused_variable_warning(is_contiguous);
182 
183  it = view.begin();
184  it = view.end();
185  rit = view.rbegin();
186  rit = view.rend();
187 
188  reference r1 = view[d];
189  ignore_unused_variable_warning(r1); // 1D access
190  reference r2 = view(p);
191  ignore_unused_variable_warning(r2); // 2D access
192 
193  // get 1-D iterator of any dimension at a given pixel location
194  first_it_type fi = view.template axis_iterator<0>(p);
195  ignore_unused_variable_warning(fi);
196  last_it_type li = view.template axis_iterator<N - 1>(p);
197  ignore_unused_variable_warning(li);
198 
199  using deref_t = PixelDereferenceAdaptorArchetype<typename View::value_type>;
200  using dtype = typename View::template add_deref<deref_t>::type;
201  }
202  View view;
203 };

268  y_iterator yit;

270

272 
273  xy_locator lc = view.xy_at(xd, yd);
274  lc = view.xy_at(d);
275 
276  typename View::reference r = view(xd, yd);

278  xd = view.width();
279  yd = view.height();
280 
281  xit = view.x_at(d);
