43 template <typename T, typename IntegralTypes, std::size_t MinSize>
44 struct promote_to_larger {
45  using current_type = boost::mp11::mp_first<IntegralTypes>;
46  using list_after_front = boost::mp11::mp_rest<IntegralTypes>;
47 
48  using type = typename std::conditional<
49  (bit_size<current_type>::value >= MinSize), current_type,
50  typename promote_to_larger<T, list_after_front, MinSize>::type>::type;
51 };

97 template <typename T, bool PromoteUnsignedToUnsigned = false,
98  bool UseCheckedInteger = false,
99  bool IsIntegral = std::is_integral<T>::value>
100 class promote_integral {
101 private:
102  static bool const is_unsigned = std::is_unsigned<T>::value;
103 
104  using bit_size_type = detail::promote_integral::bit_size<T>;
105 
106  // Define the minimum size (in bits) needed for the promoted type
107  // If T is the input type and P the promoted type, then the
108  // minimum number of bits for P are (below b stands for the number
109  // of bits of T):
110  // * if T is unsigned and P is unsigned: 2 * b
111  // * if T is signed and P is signed: 2 * b - 1
112  // * if T is unsigned and P is signed: 2 * b + 1
113  using min_bit_size_type = typename std::conditional<
114  (PromoteUnsignedToUnsigned && is_unsigned),
115  std::integral_constant<std::size_t, (2 * bit_size_type::value)>,
116  typename std::conditional<
117  is_unsigned,
118  std::integral_constant<std::size_t, (2 * bit_size_type::value + 1)>,
119  std::integral_constant<std::size_t,
120  (2 * bit_size_type::value - 1)>>::type>::type;
121 
122  // Define the list of signed integral types we are going to use
123  // for promotion
124  using signed_integral_types = boost::mp11::mp_list<short, int, long
125 #if defined(BOOST_HAS_LONG_LONG)
126  ,
127  boost::long_long_type
128 #endif
129  >;
130 
131  // Define the list of unsigned integral types we are going to use
132  // for promotion
133  using unsigned_integral_types =
134  boost::mp11::mp_list<unsigned short, unsigned int, unsigned long,
135  std::size_t
136 #if defined(BOOST_HAS_LONG_LONG)
137  ,
138  boost::ulong_long_type
139 #endif
140  >;
141 
142  // Define the list of integral types that will be used for
143  // promotion (depending in whether we was to promote unsigned to
144  // unsigned or not)
145  using integral_types =
146  typename std::conditional<(is_unsigned && PromoteUnsignedToUnsigned),
147  unsigned_integral_types,
148  signed_integral_types>::type;
149 
150 public:
151  using type = typename detail::promote_integral::promote_to_larger<
152  T, integral_types, min_bit_size_type::value>::type;
153 };
