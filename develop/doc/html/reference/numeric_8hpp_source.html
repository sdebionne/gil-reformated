107 inline void generate_gaussian_kernel(boost::gil::gray32f_view_t dst, double sigma)
108 {
109  if (dst.width() != dst.height() || dst.width() % 2 != 1)
110  throw std::invalid_argument("kernel dimensions should be odd and equal");
111 
112  const double denominator = 2 * boost::gil::pi * sigma * sigma;
113  const auto middle = boost::gil::point_t(dst.width() / 2, dst.height() / 2);
114  for (boost::gil::gray32f_view_t::coord_t y = 0; y < dst.height(); ++y)
115  {
116  for (boost::gil::gray32f_view_t::coord_t x = 0; x < dst.width(); ++x)
117  {
118  const auto delta_x = std::abs(middle.x - x);
119  const auto delta_y = std::abs(middle.y - y);
120  const double power = (delta_x * delta_x + delta_y * delta_y) / (2 * sigma * sigma);
121  const double nominator = std::exp(-power);
122  const float value = nominator / denominator;
123  dst(x, y).at(std::integral_constant<int, 0>{}) = value;
124  }
125  }
126 }
