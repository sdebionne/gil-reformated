137  std::is_arithmetic<D>::value,
138  point<typename detail::std_common_type<T, D>::type>>::type {
139  static_assert(std::is_arithmetic<D>::value,
140  "denominator is not arithmetic type");

142  if (d < 0 || 0 < d) {

144  double const y = static_cast<double>(p.y) / static_cast<double>(d);
145  return point<result_type>{static_cast<result_type>(iround(x)),
