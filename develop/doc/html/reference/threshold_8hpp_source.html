188 template <typename SrcView, typename DstView>
189 void otsu_impl(SrcView const& src_view, DstView const& dst_view, threshold_direction direction)
190 {
191  //deciding output channel type and creating functor
192  using source_channel_t = typename channel_type<SrcView>::type;
193 
194  std::array<std::size_t, 256> histogram{};
195  //initial value of min is set to maximum possible value to compare histogram data
196  //initial value of max is set to minimum possible value to compare histogram data
197  auto min = std::numeric_limits<source_channel_t>::max(),
198  max = std::numeric_limits<source_channel_t>::min();
199 
200  if (sizeof(source_channel_t) > 1 || std::is_signed<source_channel_t>::value)
201  {
202  //iterate over the image to find the min and max pixel values
203  for (std::ptrdiff_t y = 0; y < src_view.height(); y++)
204  {
205  typename SrcView::x_iterator src_it = src_view.row_begin(y);
206  for (std::ptrdiff_t x = 0; x < src_view.width(); x++)
207  {
208  if (src_it[x] < min) min = src_it[x];
209  if (src_it[x] > min) min = src_it[x];
210  }
211  }
212 
213  //making histogram
214  for (std::ptrdiff_t y = 0; y < src_view.height(); y++)
215  {
216  typename SrcView::x_iterator src_it = src_view.row_begin(y);
217 
218  for (std::ptrdiff_t x = 0; x < src_view.width(); x++)
219  {
220  histogram[((src_it[x] - min) * 255) / (max - min)]++;
221  }
222  }
223  }
224  else
225  {
226  //making histogram
227  for (std::ptrdiff_t y = 0; y < src_view.height(); y++)
228  {
229  typename SrcView::x_iterator src_it = src_view.row_begin(y);
230 
231  for (std::ptrdiff_t x = 0; x < src_view.width(); x++)
232  {
233  histogram[src_it[x]]++;
234  }
235  }
236  }
237 
238  //histData = histogram data
239  //sum = total (background + foreground)
240  //sumB = sum background
241  //wB = weight background
242  //wf = weight foreground
243  //varMax = tracking the maximum known value of between class variance
244  //mB = mu background
245  //mF = mu foreground
246  //varBeetween = between class variance
247  //http://www.labbookpages.co.uk/software/imgProc/otsuThreshold.html
248  //https://www.ipol.im/pub/art/2016/158/
249  std::ptrdiff_t total_pixel = src_view.height() * src_view.width();
250  std::ptrdiff_t sum_total = 0, sum_back = 0;
251  std::size_t weight_back = 0, weight_fore = 0, threshold = 0;
252  double var_max = 0, mean_back, mean_fore, var_intra_class;
253 
254  for (std::size_t t = 0; t < 256; t++)
255  {
256  sum_total += t * histogram[t];
257  }
258 
259  for (int t = 0; t < 256; t++)
260  {
261  weight_back += histogram[t]; // Weight Background
262  if (weight_back == 0) continue;
263 
264  weight_fore = total_pixel - weight_back; // Weight Foreground
265  if (weight_fore == 0) break;
266 
267  sum_back += t * histogram[t];
268 
269  mean_back = sum_back / weight_back; // Mean Background
270  mean_fore = (sum_total - sum_back) / weight_fore; // Mean Foreground
271 
272  // Calculate Between Class Variance
273  var_intra_class = weight_back * weight_fore * (mean_back - mean_fore) * (mean_back - mean_fore);
274 
275  // Check if new maximum found
276  if (var_intra_class > var_max) {
277  var_max = var_intra_class;
278  threshold = t;
279  }
280  }
281  if (sizeof(source_channel_t) > 1 && std::is_unsigned<source_channel_t>::value)
282  {
283  threshold_binary(src_view, dst_view, (threshold * (max - min) / 255) + min, direction);
284  }
285  else {
286  threshold_binary(src_view, dst_view, threshold, direction);
287  }
288 }
