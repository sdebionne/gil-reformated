117 template <typename Loc, typename XIterator,
118  typename YIterator> // The concrete subclass, the X-iterator and the
119  // Y-iterator
120 class pixel_2d_locator_base {
121 public:
122  using x_iterator = XIterator;
123  using y_iterator = YIterator;
124 
125  // aliasesrequired by ConstRandomAccessNDLocatorConcept
126  static const std::size_t num_dimensions = 2;
127  using value_type = typename std::iterator_traits<x_iterator>::value_type;
128  using reference =
129  typename std::iterator_traits<x_iterator>::reference; // result of
130  // dereferencing
131  using coord_t = typename std::iterator_traits<x_iterator>::
132  difference_type; // 1D difference type (same for all dimensions)
133  using difference_type =
134  point<coord_t>; // result of operator-(locator,locator)
135  using point_t = difference_type;
136  template <std::size_t D> struct axis {
137  using coord_t = typename detail::locator_axis<D, Loc>::coord_t;
138  using iterator = typename detail::locator_axis<D, Loc>::iterator;
139  };
140 
141  // aliases required by ConstRandomAccess2DLocatorConcept
142  using x_coord_t = typename point_t::template axis<0>::coord_t;
143  using y_coord_t = typename point_t::template axis<1>::coord_t;
144 
145  bool operator!=(const Loc &p) const { return !(concrete() == p); }
146 
147  x_iterator x_at(x_coord_t dx, y_coord_t dy) const {
148  Loc tmp = concrete();
149  tmp += point_t(dx, dy);
150  return tmp.x();
151  }
152  x_iterator x_at(const difference_type &d) const {
153  Loc tmp = concrete();
154  tmp += d;
155  return tmp.x();
156  }
157  y_iterator y_at(x_coord_t dx, y_coord_t dy) const {
158  Loc tmp = concrete();
159  tmp += point_t(dx, dy);
160  return tmp.y();
161  }
162  y_iterator y_at(const difference_type &d) const {
163  Loc tmp = concrete();
164  tmp += d;
165  return tmp.y();
166  }
167  Loc xy_at(x_coord_t dx, y_coord_t dy) const {
168  Loc tmp = concrete();
169  tmp += point_t(dx, dy);
170  return tmp;
171  }
172  Loc xy_at(const difference_type &d) const {
173  Loc tmp = concrete();
174  tmp += d;
175  return tmp;
176  }
177 
178  template <std::size_t D> typename axis<D>::iterator &axis_iterator() {
179  return detail::locator_axis<D, Loc>()(concrete());
180  }
181  template <std::size_t D>
182  typename axis<D>::iterator const &axis_iterator() const {
183  return detail::locator_axis<D, Loc>()(concrete());
184  }
185  template <std::size_t D>
186  typename axis<D>::iterator axis_iterator(const point_t &p) const {
187  return detail::locator_axis<D, Loc>()(concrete(), p);
188  }
189 
190  reference operator()(x_coord_t dx, y_coord_t dy) const {
191  return *x_at(dx, dy);
192  }
193  reference operator[](const difference_type &d) const {
194  return *x_at(d.x, d.y);
195  }
196 
197  reference operator*() const { return *concrete().x(); }
198 
199  Loc &operator+=(const difference_type &d) {
200  concrete().x() += d.x;
201  concrete().y() += d.y;
202  return concrete();
203  }
204  Loc &operator-=(const difference_type &d) {
205  concrete().x() -= d.x;
206  concrete().y() -= d.y;
207  return concrete();
208  }
209 
210  Loc operator+(const difference_type &d) const { return xy_at(d); }
211  Loc operator-(const difference_type &d) const { return xy_at(-d); }
212 
213  // Some locators can cache 2D coordinates for faster subsequent access. By
214  // default there is no caching
215  using cached_location_t = difference_type;
216  cached_location_t cache_location(const difference_type &d) const { return d; }
217  cached_location_t cache_location(x_coord_t dx, y_coord_t dy) const {
218  return difference_type(dx, dy);
219  }
220 
221 private:
222  Loc &concrete() { return (Loc &)*this; }
223  const Loc &concrete() const { return (const Loc &)*this; }
224 
225  template <typename X> friend class pixel_2d_locator;
226 };

426  // Returns the vertical distance (it2.y-it1.y) between two x_iterators given
427  // the difference of their x positions
428  std::ptrdiff_t y_distance_to(this_t const &p2, x_coord_t xDiff) const {
429  std::ptrdiff_t rowDiff =
430  memunit_distance(x(), p2.x()) - pixel_size() * xDiff;
431  BOOST_ASSERT((rowDiff % row_size()) == 0);
432  return rowDiff / row_size();
